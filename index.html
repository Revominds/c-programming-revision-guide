Topic 1 • Input, Output & Control Flow
Comparing Integers Program
Learn how to build a complete interactive program using user input, decision making, validation, functions, and loops. This topic forms the foundation of all practical C programming questions.
📄 comparing_integers.c 

// Comparing Integers Program

#include <stdio.h>

void compareNumbers(int a, int b)
{
    if(a > b)
    {
        printf("A is greater than B\n");
    }
    else if(b > a)
    {
        printf("B is greater than A\n");
    }
    else
    {
        printf("Both numbers are equal\n");
    }
}

int main()
{
    int a, b, choice;

    do
    {
        printf("\n=== Comparing Integers ===\n");

        printf("Enter first integer: ");
        if(scanf("%d", &a) != 1)
        {
            printf("Invalid input!\n");
            return 1;
        }

        printf("Enter second integer: ");
        scanf("%d", &b);

        compareNumbers(a, b);

        printf("\n1. Continue\n0. Exit\n");
        printf("Choice: ");
        scanf("%d", &choice);

    } while(choice != 0);

    return 0;
}

Student Grading System
Learn how to organize student records using structures and arrays while calculating grades, averages, and generating reports.
📄 student_grading_system.c 

// Student Grading System

#include <stdio.h>

struct Student
{
    char name[50];
    float score;
    char grade;
};

char calculateGrade(float score)
{
    if(score >= 80)
        return 'A';
    else if(score >= 70)
        return 'B';
    else if(score >= 60)
        return 'C';
    else if(score >= 50)
        return 'D';

    return 'F';
}

int main()
{
    struct Student s[3];
    int i;

    for(i = 0; i < 3; i++)
    {
        printf("\nEnter student name: ");
        scanf("%s", s[i].name);

        printf("Enter score: ");
        scanf("%f", &s[i].score);

        s[i].grade = calculateGrade(s[i].score);
    }

    printf("\n=== Student Report ===\n");

    for(i = 0; i < 3; i++)
    {
        printf("%s - %.2f - Grade %c\n",
        s[i].name,
        s[i].score,
        s[i].grade);
    }

    return 0;
}

Employee Management System
Build systems that store employee records permanently using binary files and structures.
📄 employee_management.c 

// Employee Management System

#include <stdio.h>

struct Employee
{
    int id;
    char name[50];
    float salary;
};

int main()
{
    FILE *fp;

    struct Employee emp;

    fp = fopen("employee.dat", "ab");

    if(fp == NULL)
    {
        printf("File could not open!\n");
        return 1;
    }

    printf("Enter Employee ID: ");
    scanf("%d", &emp.id);

    printf("Enter Employee Name: ");
    scanf("%s", emp.name);

    printf("Enter Salary: ");
    scanf("%f", &emp.salary);

    fwrite(&emp, sizeof(emp), 1, fp);

    printf("Employee saved successfully!\n");

    fclose(fp);

    return 0;
}

Topic 4 • Validation & Pointers
Banking System
Create secure transaction systems using pointers, validation, loops, and structures.
📄 banking_system.c 

// Banking System

#include <stdio.h>

struct Account
{
    char name[50];
    float balance;
};

void deposit(struct Account *acc, float amount)
{
    acc->balance += amount;
}

void withdraw(struct Account *acc, float amount)
{
    if(amount > acc->balance)
    {
        printf("Insufficient balance!\n");
    }
    else
    {
        acc->balance -= amount;
        printf("Withdrawal successful!\n");
    }
}

int main()
{
    struct Account acc = {"John", 1000};

    deposit(&acc, 500);
    withdraw(&acc, 200);

    printf("Current Balance: %.2f\n", acc.balance);

    return 0;
}

// Functions and Pointers

#include <stdio.h>
#include <string.h>

int factorial(int n)
{
    if(n == 0)
        return 1;

    return n * factorial(n - 1);
}

void swap(int *a, int *b)
{
    int temp;

    temp = *a;
    *a = *b;
    *b = temp;
}

int main()
{
    int x = 10;
    int y = 20;

    printf("Before Swap: %d %d\n", x, y);

    swap(&x, &y);

    printf("After Swap: %d %d\n", x, y);

    printf("Factorial of 5 = %d\n", factorial(5));

    return 0;
}
